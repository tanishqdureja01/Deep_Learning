Define and Compile Keras Model (Custom CNN)¶

In this cell, a custom CNN architecture is defined in Keras, with layers for convolution, batch normalization, pooling, dropout, and dense layers.

Compiling the Model¶

  • Optimizer used during Back Propagation for weight and bias adjustment - Adam (adjusts the learning rate adaptively).
  • Loss Function used - Categorical Crossentropy (used when multiple categories/classes are present).
  • Metrics used for evaluation - Accuracy
In [4]:
# Define Keras CNN model architecture
keras_model = Sequential([
    layers.Conv2D(32, (3, 3), padding='same', activation='relu', input_shape=(32, 32, 3)),
    layers.BatchNormalization(),
    layers.MaxPooling2D(pool_size=(2, 2)),
    layers.Dropout(0.3),

    layers.Conv2D(64, (3, 3), padding='same', activation='relu'),
    layers.BatchNormalization(),
    layers.MaxPooling2D(pool_size=(2, 2)),
    layers.Dropout(0.4),

    layers.Conv2D(128, (3, 3), padding='same', activation='relu'),
    layers.BatchNormalization(),
    layers.MaxPooling2D(pool_size=(2, 2)),
    layers.Dropout(0.5),

    layers.Flatten(),
    layers.Dense(128, activation='relu'),
    layers.BatchNormalization(),
    layers.Dropout(0.5),
    layers.Dense(10, activation='softmax')
])

# Compile model with optimizer, loss, and metrics
keras_model.compile(optimizer='adam', loss=keras.losses.categorical_crossentropy, metrics=['accuracy'])
keras_model.summary()

Fitting the Model¶

  • Epochs - One epoch is one complete cycle (forward pass + backward pass).

# Train Keras model with early stopping and learning rate reduction
early_stopping = EarlyStopping(monitor='val_loss', patience=10, restore_best_weights=True)
lr_reduction = ReduceLROnPlateau(monitor='val_loss', factor=0.5, patience=5, min_lr=1e-5)

history = keras_model.fit(train_images, train_labels, batch_size=64, epochs=50, 
                          validation_data=(val_images, val_labels),
                          callbacks=[early_stopping, lr_reduction])

# Save trained model
keras_model.save("keras_custom_cnn_model.h5")

# Load ResNet-18 and modify for CIFAR-10
model = torchvision.models.resnet18(pretrained=False, num_classes=10)
device = torch.device("cuda" if torch.cuda.is_available() else "cpu")
model = model.to(device)
criterion = nn.CrossEntropyLoss()
optimizer = optim.Adam(model.parameters(), lr=0.001)

# Define PyTorch training function with tracking of training and validation loss and accuracy
def train_model(model, trainloader, testloader, epochs=20):
    training_loss, validation_loss = [], []
    training_accuracy, validation_accuracy = [], []
    
    for epoch in range(epochs):
        model.train()
        running_loss, correct, total = 0.0, 0, 0
        
        for inputs, labels in trainloader:
            inputs, labels = inputs.to(device), labels.to(device)
            optimizer.zero_grad()
            outputs = model(inputs)
            loss = criterion(outputs, labels)
            loss.backward()
            optimizer.step()
            
            # Update running loss and correct predictions count for training
            running_loss += loss.item()
            _, predicted = torch.max(outputs.data, 1)
            total += labels.size(0)
            correct += (predicted == labels).sum().item()
        
        # Calculate average training loss and accuracy for the epoch
        epoch_training_loss = running_loss / len(trainloader)
        epoch_training_accuracy = 100 * correct / total
        training_loss.append(epoch_training_loss)
        training_accuracy.append(epoch_training_accuracy)
        
        # Validation phase
        model.eval()
        val_running_loss, correct, total = 0.0, 0, 0
        with torch.no_grad():
            for inputs, labels in testloader:
                inputs, labels = inputs.to(device), labels.to(device)
                outputs = model(inputs)
                loss = criterion(outputs, labels)
                val_running_loss += loss.item()
                
                # Update correct predictions count for validation
                _, predicted = torch.max(outputs.data, 1)
                total += labels.size(0)
                correct += (predicted == labels).sum().item()
        
        # Calculate average validation loss and accuracy for the epoch
        epoch_validation_loss = val_running_loss / len(testloader)
        epoch_validation_accuracy = 100 * correct / total
        validation_loss.append(epoch_validation_loss)
        validation_accuracy.append(epoch_validation_accuracy)
        
        print(f"Epoch [{epoch + 1}/{epochs}], Training Loss: {epoch_training_loss:.4f}, "
              f"Training Accuracy: {epoch_training_accuracy:.2f}%, "
              f"Validation Loss: {epoch_validation_loss:.4f}, "
              f"Validation Accuracy: {epoch_validation_accuracy:.2f}%")
    
    return training_loss, training_accuracy, validation_loss, validation_accuracy

# Train PyTorch model and save it
train_loss, train_acc, val_loss, val_acc = train_model(model, trainloader, testloader, epochs=20)
torch.save(model.state_dict(), "pytorch_resnet18_cifar10.pth")

def evaluate_model(model, testloader):
    model.eval()
    correct, total = 0, 0
    with torch.no_grad():
        for inputs, labels in testloader:
            inputs, labels = inputs.to(device), labels.to(device)
            outputs = model(inputs)

            total += labels.size(0)
            correct += (predicted == labels).sum().item()
    accuracy = 100 * correct / total
    print(f'PyTorch Model - Test Accuracy: {accuracy:.2f}%')
